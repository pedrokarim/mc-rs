Configuration

Server configuration via server.toml with sensible defaults.

The server creates a default server.toml on first run. All settings are optional — defaults are applied for any missing value.

[server] Section

KeyTypeDefaultDescription
addressString"0.0.0.0"Bind address
portu1619132UDP port
motdString"MC-RS Server"Server message of the day (shown in server list)
max_playersu3220Maximum concurrent players
online_modebooltrueVerify Xbox Live authentication
gamemodeString"survival"Default gamemode (survival, creative, adventure)
difficultyString"normal"Difficulty (peaceful, easy, normal, hard)
view_distanceu3210Maximum view distance in chunks
tick_rateu3220Server tick rate (ticks per second)
operator_permission_levelu322Default permission level for operators

[world] Section

KeyTypeDefaultDescription
nameString"world"World directory name
generatorString"default"World generator (default, flat, void)
seedi64randomWorld generation seed
spawn_xi320Default spawn X position
spawn_yi3270Default spawn Y position
spawn_zi320Default spawn Z position
auto_save_intervalu64300Auto-save interval in seconds
max_build_heighti32319Maximum build height
enable_netherbooltrueEnable Nether dimension
enable_endbooltrueEnable End dimension

[logging] Section

KeyTypeDefaultDescription
levelString"info"Log level (trace, debug, info, warn, error)
fileString""Log file path (empty = stdout only)
formatString"pretty"Log format (pretty, json, compact)

[permissions] Section

KeyTypeDefaultDescription
enforce_whitelistboolfalseOnly whitelisted players can join
ops_fileString"ops.json"Path to operators file
whitelist_fileString"whitelist.json"Path to whitelist file
banned_players_fileString"banned-players.json"Path to banned players file
banned_ips_fileString"banned-ips.json"Path to banned IPs file

[rcon] Section

KeyTypeDefaultDescription
enabledboolfalseEnable RCON
portu1625575RCON TCP port
passwordString""RCON password (required if enabled)

[query] Section

KeyTypeDefaultDescription
enabledboolfalseEnable GameSpy4 query
portu1619132Query UDP port

[packs] Section

KeyTypeDefaultDescription
behavior_packsVec<String>[]List of behavior pack paths to load

Example Configuration

# MC-RS Server Configuration

[server]
address = "0.0.0.0"
port = 19132
motd = "My MC-RS Server"
max_players = 20
online_mode = true
gamemode = "survival"
difficulty = "normal"
view_distance = 10
tick_rate = 20
operator_permission_level = 2

[world]
name = "world"
generator = "default"
# seed = 12345  # omit for random seed
spawn_x = 0
spawn_y = 70
spawn_z = 0
auto_save_interval = 300
max_build_height = 319
enable_nether = true
enable_end = true

[logging]
level = "info"
file = ""
format = "pretty"

[permissions]
enforce_whitelist = false
ops_file = "ops.json"
whitelist_file = "whitelist.json"
banned_players_file = "banned-players.json"
banned_ips_file = "banned-ips.json"

[rcon]
enabled = false
port = 25575
password = ""

[query]
enabled = false
port = 19132

[packs]
behavior_packs = []
